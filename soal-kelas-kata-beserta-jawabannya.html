Jawab:
= 60

Penjelasan:
Keterangan
ft = frekuensi total, fr = frek. reguler
fp = frek. paralel, nr = jml nilai reguler
np = jml nilai paralel, mt = rata² total
mr = rata² reguler, mp = rata² paralel

Dik :
ft = fr + fp = 30 + 10 = 40
---> ft = 40
nr/fr = 4 + np/fp
nr/30 = 4 + np/10
nr/30 = 120/30 + 3np/30
3np = nr - 120
---> np = ⅓(nr - 120)
mt = 59 = (nr+np)/(ft)
mt = 59 = (nr+⅓(nr - 120))/40
mt = 59 = (⅓(3nr)+⅓(nr - 120))/40
mt = 59 = (3nr+nr-120)/(40(3))
mt = 59 = (4nr-120)/120
4nr - 120 = 120(59)
